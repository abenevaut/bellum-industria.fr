# GitLab

## GitLab-CI

### Installation sur Mac OS

	brew update
	brew install rbenv
	brew install ruby-build
	brew install openssl
	brew install gcc
	CC=gcc RUBY_CONFIGURE_OPTS="--with-openssl-dir=`brew --prefix openssl` --with-readline-dir=`brew --prefix readline` --with-gcc=gcc --enable-shared" rbenv install 2.0.0-p353
	echo 'if which rbenv > /dev/null; then eval "$(rbenv init -)"; fi' >> ~/.profile
	rbenv global 2.0.0-p353


	cd ~/
	git clone https://gitlab.com/gitlab-org/gitlab-ci-runner.git
	cd gitlab-ci-runner
	git checkout VERSION_YOU_NEED # Ex. v5.0.0



	gem install bundler

	bundle install --deployment





	https://ci.gitlab.com
