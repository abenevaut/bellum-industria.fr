# Install Git & SVN servers


http://odyniec.net/articles/ubuntu-subversion-server/

$ sudo apt-get install subversion
$ sudo mkdir /var/svn
$ sudo groupadd svn
$ sudo chgrp svn /var/svn
$ sudo chmod g+w /var/svn
$ sudo chmod g+s /var/svn
$ sudo adduser USERNAME svn

$ svnadmin create /var/svn/depo_one
$ sudo chmod -R g+w /var/svn/depo_one
$ sudo chmod -R g+s /var/svn/depo_one

----------

$ sudo apt-get install git-core
$ sudo mkdir /var/git
$ sudo groupadd git
$ sudo chgrp git /var/git
$ sudo chmod g+w /var/git
$ sudo chmod g+s /var/git
$ sudo adduser USERNAME git

$ sudo mkdir repo_one && cd repo_one && sudo git --bare init
$ sudo chmod -R g+w /var/git/repo_one
$ sudo chmod -R g+s /var/git/repo_one


# Git commands

Get change from 10 last commits

	git diff --name-only HEAD~10..HEAD

Get change since 2 last days commits

	git diff --stat @{2.days.ago}

Cancel a merge

	git reset --merge ORIG_HEAD

Update repo information from remote

	git remote update --prune
